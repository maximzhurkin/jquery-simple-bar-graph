jQuery Simple Bar Graph
Very simple plugin for generating bar charts based on key/value data
Download
Github
Usage

Include jquery.simple-bar-graph.css to head

<link href="/dist/css/jquery.simple-bar-graph.min.css" rel="stylesheet">

Include jQuery and SimpleBarGraph

<script src="http://code.jquery.com/jquery-latest.min.js"></script>
<script src="/dist/js/jquery.simple-bar-graph.min.js"></script>

Initialize plugin

$(document).ready(function(){
    $('#graph').simpleBarGraph({
        data: [
            { key: 'Monday', value: 120 },
            { key: 'Tuesday', value: 96 },
            { key: 'Wednesday', value: 46 },
            { key: 'Thursday', value: 58 },
            { key: 'Friday', value: 64 },
            { key: 'Saturday', value: 108 },
            { key: 'Sunday', value: 91 }
        ]
    });
});

Custom settings

$('#graph').simpleBarGraph({
    data: [],
    rowsCount: 5,
    height: '200px',
    rowCaptionsWidth: '16px',
    barsColor: '#C496CF',
    popups: true
    delayAnimation: 15 //ms
});
